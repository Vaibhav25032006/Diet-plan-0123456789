import React from "react";
import { Clock, CheckSquare, Lock, Eye, AlertCircle, PlayCircle, Star, Moon, CalendarDays, Ban } from "lucide-react";
import { Task, PlanType } from "../types";

interface TaskTrackerProps {
  tasks: Task[];
  planNameEn: string;
  planNameHi: string;
  targetEn: string;
  targetHi: string;
  completedTasks: Record<string, { completed: boolean }>;
  simulatedHour: number;
  simulatedMinute: number;
  onAdjustTime: (h: number, m: number) => void;
  onVerifyWithCamera: (task: Task) => void;
  onDirectTick: (task: Task) => void;
  speakText: (text: string) => void;
  guidelinesEn: string[];
  guidelinesHi: string[];
  forbiddenEn: string[];
  forbiddenHi: string[];
  isNightMode?: boolean;
  use24HourFormat?: boolean;
}

export default function TaskTracker({
  tasks,
  planNameEn,
  planNameHi,
  targetEn,
  targetHi,
  completedTasks,
  simulatedHour,
  simulatedMinute,
  onAdjustTime,
  onVerifyWithCamera,
  onDirectTick,
  speakText,
  guidelinesEn,
  guidelinesHi,
  forbiddenEn,
  forbiddenHi,
  isNightMode = true,
  use24HourFormat = false,
}: TaskTrackerProps) {
  
  // Clean zero pad time
  const formatTimeToken = (num: number) => String(num).padStart(2, "0");

  const formatHourMin = (totalMins: number) => {
    const h = Math.floor(totalMins / 60) % 24;
    const m = totalMins % 60;
    if (use24HourFormat) {
      return `${String(h).padStart(2, "0")}:${String(m).padStart(2, "0")}`;
    }
    const isPm = h >= 12;
    const displayH = h % 12 === 0 ? 12 : h % 12;
    const displayM = String(m).padStart(2, "0");
    const labelAmPm = isPm ? "PM" : "AM";
    return `${String(displayH).padStart(2, "0")}:${displayM} ${labelAmPm}`;
  };

  const getTaskStatus = (task: Task) => {
    // Math minutes
    const taskMin = task.hour * 60 + task.minute;
    const currentMin = simulatedHour * 60 + simulatedMinute;

    const isDone = !!completedTasks[task.id]?.completed;
    if (isDone) {
      return { state: "COMPLETED", message: "Task Verified / पूरा हो चुका है" };
    }

    const startWindow = taskMin;
    const endWindow = taskMin + 120; // Exactly 2 hours window (120 minutes)

    const startTimeFormatted = formatHourMin(startWindow);
    const endTimeFormatted = formatHourMin(endWindow);

    if (currentMin < startWindow) {
      return { 
        state: "LOCKED_FUTURE", 
        message: `Locked: Opens at ${startTimeFormatted}`,
        subMessage: `यह कार्य ठीक ${startTimeFormatted} पर खुलेगा` 
      };
    } else if (currentMin > endWindow) {
      return { 
        state: "LOCKED_EXPIRED", 
        message: "Locked (Expired)",
        subMessage: `समय सीमा समाप्त (यह कार्य ${startTimeFormatted} से ${endTimeFormatted} तक खुला था)` 
      };
    } else {
      return { 
        state: "UNLOCKED", 
        message: "Open for Verification / खुला है", 
        subMessage: `Closes in ${endWindow - currentMin} mins (${endTimeFormatted} बजे तक खुला रहेगा)` 
      };
    }
  };

  const handleHearInstructions = (task: Task) => {
    const speech = `कार्य का नाम: ${task.titleHi}. विवरण: ${task.descriptionHi}. ${
      task.isHerbalifeProduct ? `कृपया ध्यान दें, इसके लिए हर्बालाइफ ${task.requiredProduct} आवश्यक है.` : ""
    }`;
    speakText(speech);
  };

  return (
    <div className="space-y-6" id="task-tracker-container">

      {/* Hero Banner Area */}
      <div className={`border rounded-2xl p-6 shadow-md relative overflow-hidden transition-all ${
        isNightMode ? "bg-slate-900 border-slate-800" : "bg-white border-slate-200"
      }`}>
        <div className="absolute -right-10 -bottom-10 opacity-5">
          <Clock className={`w-48 h-48 ${isNightMode ? "text-white" : "text-slate-900"}`} />
        </div>
        
        <div className="relative space-y-2">
          <div className="flex gap-2 items-center">
            <span className="text-[10px] bg-emerald-500/10 text-emerald-500 py-0.5 px-2 rounded-full border border-emerald-500/20 font-mono tracking-wider font-bold">
              CURRENT FOCUS
            </span>
          </div>
          <h2 className={`text-xl sm:text-2xl font-black leading-tight font-sans ${
            isNightMode ? "text-white" : "text-slate-900"
          }`}>
            {planNameEn} <span className={`block sm:inline font-medium text-sm ${
              isNightMode ? "text-slate-400" : "text-slate-500"
            }`}>({planNameHi})</span>
          </h2>
          <p className={`text-xs font-semibold ${isNightMode ? "text-slate-300" : "text-slate-700"}`}>
            Target: {targetEn}
          </p>
          <p className={`text-[11px] font-sans italic ${isNightMode ? "text-slate-400" : "text-slate-500"}`}>
            लक्ष्य: {targetHi}
          </p>
        </div>
      </div>

      {/* 2-Hour Window Simulated Slider Panel */}
      <div className={`border p-5 rounded-2xl space-y-3 shadow-inner transition-all ${
        isNightMode ? "bg-slate-950 border-amber-500/15" : "bg-amber-500/5 border-amber-500/20"
      }`}>
        <div className="flex flex-col sm:flex-row sm:items-center justify-between gap-1">
          <div className="flex items-center space-x-2">
            <span className="p-1 px-1.5 bg-amber-500/10 text-amber-500 rounded border border-amber-500/20 text-xs font-mono font-bold">
              SIMULATOR
            </span>
            <h3 className={`text-xs font-bold font-mono uppercase tracking-wider ${
              isNightMode ? "text-slate-300" : "text-slate-800"
            }`}>
              Time-Travel Clock Controls (परीक्षण मोड)
            </h3>
          </div>
          <span className={`text-xs font-mono font-bold ${
            isNightMode ? "text-amber-300" : "text-amber-700"
          }`}>
            Simulated Hour: {formatTimeToken(simulatedHour)}:{formatTimeToken(simulatedMinute)}
          </span>
        </div>
        <p className={`text-[10px] leading-normal ${isNightMode ? "text-slate-400" : "text-slate-600"}`}>
          Adjust the slider to simulate different times of the day! This allows you to verify the strict <strong>2-hour lock</strong> rules (tasks are only unlockable within 2 hours of their scheduled time).
        </p>

        <div className="space-y-2">
          <div className="flex items-center space-x-4">
            <span className={`text-xs font-mono w-14 ${isNightMode ? "text-slate-400" : "text-slate-600"}`}>Hour [0-23]</span>
            <input
              type="range"
              min="0"
              max="23"
              value={simulatedHour}
              onChange={(e) => onAdjustTime(parseInt(e.target.value), simulatedMinute)}
              className="flex-1 accent-amber-500 h-1 bg-slate-800 rounded-lg appearance-none cursor-ew-resize"
              id="time-hour-slider"
            />
          </div>
          <div className="flex items-center space-x-4">
            <span className={`text-xs font-mono w-14 ${isNightMode ? "text-slate-400" : "text-slate-600"}`}>Min [0-59]</span>
            <input
              type="range"
              min="0"
              max="59"
              value={simulatedMinute}
              onChange={(e) => onAdjustTime(simulatedHour, parseInt(e.target.value))}
              className="flex-1 accent-amber-500 h-1 bg-slate-800 rounded-lg appearance-none cursor-ew-resize"
              id="time-minute-slider"
            />
          </div>
        </div>
      </div>

      {/* Tasks Queue List */}
      <div className="space-y-4">
        <h3 className={`text-xs font-bold font-mono uppercase tracking-widest pl-1 ${
          isNightMode ? "text-slate-400" : "text-slate-500"
        }`}>
          Today's Timeline Tasks Schedule
        </h3>

        <div className="space-y-3">
          {tasks.map((task) => {
            const { state, message, subMessage } = getTaskStatus(task);
            const isDone = state === "COMPLETED";
            const isLockedFuture = state === "LOCKED_FUTURE";
            const isExpired = state === "LOCKED_EXPIRED";
            const isUnlocked = state === "UNLOCKED";

            // Light / Dark card styling mappings
            let borderClass = "";
            let timeColor = "";
            
            if (isNightMode) {
              if (isDone) {
                borderClass = "border-emerald-500/20 bg-emerald-500/5 hover:border-emerald-500/40 opacity-100";
                timeColor = "text-emerald-400 bg-emerald-500/10 border-emerald-500/20";
              } else if (isUnlocked) {
                borderClass = "border-amber-500/40 bg-slate-900 ring-1 ring-amber-500/10 hover:border-amber-400 opacity-100 shadow-md";
                timeColor = "text-amber-400 bg-amber-500/10 border-amber-500/20";
              } else {
                borderClass = "border-slate-800 bg-slate-900/40 opacity-70";
                timeColor = "text-slate-500 bg-slate-800 border-slate-700/50";
              }
            } else {
              // Light mode
              if (isDone) {
                borderClass = "border-emerald-200 bg-emerald-50/40 opacity-100 hover:border-emerald-300 shadow-xs";
                timeColor = "text-emerald-700 bg-emerald-100/50 border-emerald-200";
              } else if (isUnlocked) {
                borderClass = "border-amber-400 bg-amber-50/20 border-2 font-semibold opacity-100 shadow-inner";
                timeColor = "text-amber-700 bg-amber-100 border-amber-200";
              } else {
                borderClass = "border-slate-200 bg-white opacity-60";
                timeColor = "text-slate-500 bg-slate-100 border-slate-200";
              }
            }

            return (
              <div
                key={task.id}
                className={`border rounded-2xl p-4 transition-all flex flex-col sm:flex-row sm:items-center justify-between gap-4 ${borderClass}`}
                id={`task-card-${task.id}`}
              >
                {/* Left Description area */}
                <div className="flex items-start space-x-3.5">
                  <div className={`p-2.5 rounded-xl text-xs font-bold font-mono border ${timeColor}`}>
                    {task.time}
                  </div>
                  
                  <div className="space-y-1">
                    <div className="flex items-center flex-wrap gap-1.5">
                      <h4 className={`text-sm font-bold tracking-tight ${
                        isNightMode ? "text-white" : "text-slate-900"
                      }`}>{task.titleEn}</h4>
                      {task.isHerbalifeProduct && (
                        <span className="text-[9px] bg-indigo-550/10 text-indigo-500 py-0.5 px-2 rounded-full border border-indigo-500/30 uppercase tracking-white font-bold">
                          Herbalife Product: {task.requiredProduct}
                        </span>
                      )}
                    </div>
                    <p className={`text-xs leading-normal ${
                      isNightMode ? "text-slate-300" : "text-slate-700"
                    }`}>{task.descriptionEn}</p>
                    <p className={`text-[11px] font-sans leading-relaxed italic ${
                      isNightMode ? "text-slate-400" : "text-slate-500"
                    }`}>
                      हिंदी: {task.descriptionHi}
                    </p>
                  </div>
                </div>

                {/* Right Verification Action Buttons Area */}
                <div className={`flex sm:flex-col items-start sm:items-end justify-between sm:justify-center border-t sm:border-0 pt-3 sm:pt-0 gap-2 ${
                  isNightMode ? "border-slate-800/60" : "border-slate-200"
                }`}>
                  
                  {/* Lock Info / Duration tags */}
                  <div className="text-left sm:text-right">
                    <p className={`text-[10px] font-mono leading-none ${
                        isDone ? "text-emerald-500" : isUnlocked ? "text-amber-500 font-bold" : "text-slate-500"
                    }`}>
                      {message}
                    </p>
                    {subMessage && (
                      <p className={`text-[9px] font-sans italic mt-1 leading-none ${
                        isNightMode ? "text-slate-400" : "text-slate-600"
                      }`}>{subMessage}</p>
                    )}
                  </div>

                  <div className="flex items-center space-x-2">
                    
                    {/* Listen instruction button */}
                    <button
                      type="button"
                      onClick={() => handleHearInstructions(task)}
                      className={`cursor-pointer p-2 rounded-lg text-xs transition-colors border ${
                        isNightMode
                          ? "bg-slate-800 border-slate-700 text-slate-300 hover:bg-slate-705"
                          : "bg-slate-100 border-slate-200 text-slate-700 hover:bg-slate-200"
                      }`}
                      title="Speak Instruction (Hindi)"
                    >
                      <Eye className="w-3.5 h-3.5" />
                    </button>

                    {/* Completion actions */}
                    {isUnlocked && (
                      <>
                        {(task.noCameraNeeded || task.id.toLowerCase().includes("sleep") || task.titleEn.toLowerCase().includes("sleep")) ? (
                          <button
                            type="button"
                            onClick={() => onDirectTick(task)}
                            className="cursor-pointer bg-amber-500 hover:bg-amber-600 text-slate-950 font-bold text-xs px-3.5 py-1.5 rounded-lg flex items-center gap-1 shadow-sm transition-colors"
                          >
                            <Moon className="w-3.5 h-3.5" />
                            Direct Tick
                          </button>
                        ) : (
                          <button
                            type="button"
                            onClick={() => onVerifyWithCamera(task)}
                            className="cursor-pointer bg-gradient-to-r from-emerald-500 to-teal-500 hover:from-emerald-600 hover:to-teal-600 text-slate-950 font-bold text-xs px-3.5 py-1.5 rounded-lg flex items-center gap-1 shadow-md transition-colors animate-pulse"
                          >
                            Verify API Camera
                          </button>
                        )}
                      </>
                    )}

                    {isLockedFuture && (
                      <span className={`p-1 px-2 text-[10px] font-mono rounded border flex items-center gap-1 leading-none ${
                        isNightMode 
                          ? "bg-slate-800/80 text-slate-500 border-slate-700" 
                          : "bg-slate-205/60 text-slate-500 border-slate-200"
                      }`}>
                        <Lock className="w-3 h-3" /> Lock
                      </span>
                    )}

                    {isExpired && (
                      <span className="p-1 px-2 bg-red-500/10 text-red-500 text-[10px] font-mono rounded border border-red-550/20 flex items-center gap-1 leading-none">
                        <Ban className="w-3 h-3" /> Expired
                      </span>
                    )}
                    
                    {isDone && (
                      <span className="p-1 px-2 bg-emerald-500/10 text-emerald-500 text-[10px] font-mono rounded border border-emerald-500/20 flex items-center gap-1 font-bold leading-none">
                        <CheckSquare className="w-3 h-3" /> Cleared
                      </span>
                    )}

                  </div>

                </div>

              </div>
            );
          })}
        </div>
      </div>

      {/* Guidelines & Avoid Area Cards */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-6 pt-4">
        
        {/* Do's Guidelines */}
        <div className={`border rounded-2xl p-5 space-y-3 transition-colors ${
          isNightMode ? "bg-slate-900 border-slate-800" : "bg-emerald-500/5 border-emerald-500/10"
        }`}>
          <h4 className={`text-xs font-bold font-mono uppercase tracking-widest border-b pb-2 ${
            isNightMode ? "text-emerald-400 border-emerald-500/10" : "text-emerald-700 border-emerald-500/20"
          }`}>
            Daily Habits & Guidelines (क्या खाएं)
          </h4>
          <ul className="space-y-2">
            {guidelinesEn.map((item, idx) => (
              <li key={idx} className="text-xs flex items-start gap-2">
                <span className="text-emerald-500 font-bold">✔</span>
                <div>
                  <p className={`font-sans leading-tight ${isNightMode ? "text-slate-300" : "text-slate-800"}`}>{item}</p>
                  <p className={`text-[10px] italic mt-0.5 ${isNightMode ? "text-slate-400" : "text-slate-500"}`}>{guidelinesHi[idx]}</p>
                </div>
              </li>
            ))}
          </ul>
        </div>

        {/* Don'ts Forbidden */}
        <div className={`border rounded-2xl p-5 space-y-3 transition-colors ${
          isNightMode ? "bg-slate-900 border-slate-800" : "bg-red-500/5 border-red-500/10"
        }`}>
          <h4 className={`text-xs font-bold font-mono uppercase tracking-widest border-b pb-2 ${
            isNightMode ? "text-red-400 border-red-500/10" : "text-red-600 border-red-500/20"
          }`}>
            Foods to Avoid (इन चीजों से बचें)
          </h4>
          <ul className="space-y-2">
            {forbiddenEn.map((item, idx) => (
              <li key={idx} className="text-xs flex items-start gap-2">
                <span className="text-red-500 font-bold">✘</span>
                <div>
                  <p className={`font-sans leading-tight ${isNightMode ? "text-slate-300" : "text-slate-800"}`}>{item}</p>
                  <p className={`text-[10px] italic mt-0.5 ${isNightMode ? "text-slate-400" : "text-slate-500"}`}>{forbiddenHi[idx]}</p>
                </div>
              </li>
            ))}
          </ul>
        </div>

      </div>

    </div>
  );
}
